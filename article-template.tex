% Dokumentenklasse "article", DIN A4, Schriftgröße 12pt
\documentclass[a4paper,12pt]{article}

% T1-Kodierung, damit Umlaute als gesonderte Zeichen
% dargestellt werden können.
\usepackage[T1]{fontenc}

% Zeichensatzkodierung Unicode (UTF-8)
\usepackage[utf8]{inputenc}

% Aus "Table of Contents" wird "Inhaltsverzeichnis";
% außerdme wird für die Trennung die neue deutsche Rechtschreibung verwendet.
\usepackage[ngerman]{babel}

% Einzug der ersten Zeile bei Beginn eines neuen Absatzes:
\setlength{\parindent}{0em} % Einzug: {1.5em}

% Abstand zwischen zwei Absätzen:
\setlength{\parskip}{1.5ex}

% Titel, Autor und heutiges Datum
\title{title}
\author{author}
\date{\today}

% Nun beginnt das eigentliche Dokument.
\begin{document}

% Titelseite einfügen
\maketitle

% neue Seite beginnen
\newpage

% Inhaltsverzeichnis einfügen
\tableofcontents

% neue Seite beginnen
\newpage

% Kapitel 1
\section{section}

Fließtext

% Unterkapitel
\subsection{subsection}

Fließtext

\end{document}
