% Nun beginnt das eigentliche Dokument.
\begin{document}

\maketitle

% neue Seite beginnen
\newpage

% Inhaltsverzeichnis einfügen
\tableofcontents

% neue Seite beginnen
\newpage

% Kapitel 1
\section{section}

Fließtext

\subsection{subsection}
